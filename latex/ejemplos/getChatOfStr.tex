\documentclass[12pt,a4paper]{extarticle}
\usepackage[utf8]{inputenc}

\def\fl#1{\flx#1\empty\empty\empty}
\def\flx#1#2#3\empty{%
\edef\fst{#1}%
 \edef\cdar{#2}%
 \edef\cddr{#3}%
  \ifx\cddr\empty
    \let\lst\cdar
  \else
     \expandafter\flxx
   \fi
  #3}
\def\flxx#1#2\empty{%
 \edef\car{#1}%
  \ifx\car\empty
  \else
     \let\lst\car
     \expandafter\flxx
   \fi
  #2\empty}

\begin{document}

\noindent \texttt{fst(abcd): } \fl{abcd}{\fst}\\
\texttt{lst(abcd): } \fl{abcd}{\lst}\\
\texttt{lst(abc): }  \fl{abc}{\lst}\\
\texttt{lst(ab): }  \fl{ab}{\lst}\\
\texttt{lst(abcde): }  \fl{abcde}{\lst}\\

\end{document}
